def color2int(color,format)
  case color
  when Integer
    return color
  when Array
    return format.mapRGB(*color)
  else
    raise Error,'first argument must be Integer or Array'
  end
end

def convertRect(rect)
  case rect
  when NilClass
    return Rect.new( 0, 0, 0, 0 )
  when Array
    return Rect.new( rect[0], rect[1], rect[2], rect[3] )
  else
    return rect
  end
end
